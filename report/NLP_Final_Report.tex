\section{Giới thiệu}
% ===================================================================

\subsection{Bối cảnh và động lực}

Thương mại điện tử (TMĐT) tại Việt Nam đạt quy mô \textbf{25 tỷ USD năm 2024} và dự báo 45 tỷ USD vào 2025~\cite{google_ecns2024}, với Shopee, Lazada, TikTok Shop dẫn đầu. Mỗi sản phẩm phổ biến tích luỹ \textbf{hàng nghìn đánh giá} --- một sản phẩm trong tập dữ liệu của chúng tôi có tới 384 review. Trước khi mua, người dùng thường có câu hỏi \textbf{theo khía cạnh cụ thể} (\textit{pin, camera, giá, bảo hành}\ldots) nhưng phải tự đọc và tổng hợp thủ công hàng trăm bình luận --- chậm, phiến diện (chỉ đọc được vài review đầu), và dễ bị nhiễu bởi đánh giá giả.

Hệ thống tìm kiếm từ khoá truyền thống của sàn TMĐT không giải quyết được vì: (i) câu hỏi ngôn ngữ tự nhiên (``pin có trâu không'') không khớp từ khoá trong review (``dùng 2 ngày mới sạc''); (ii) không tổng hợp được \textbf{xu hướng} (đa số khen hay chê pin?); (iii) tiếng Việt là ngôn ngữ đơn lập, nhiều từ lóng/viết tắt/sai chính tả.

\textbf{Retrieval-Augmented Generation (RAG)}~\cite{lewis2020rag} kết hợp \textit{truy xuất thông tin} và \textit{sinh văn bản}, cho phép mô hình ngôn ngữ trả lời dựa trên \textbf{nguồn tri thức cụ thể} (review thật) thay vì ``bịa'' từ tham số. \textbf{Graph RAG}~\cite{edge2024graphrag} mở rộng bằng \textbf{đồ thị tri thức} (brand $\to$ aspect $\to$ sentiment) để bổ sung ngữ cảnh thống kê --- ví dụ ``56\% review về pin là tích cực'' --- mà truy xuất văn bản đơn thuần không nắm được.

\subsection{Phát biểu bài toán}
\label{sec:problem}

Dự án tập trung vào một bài toán cụ thể: \textbf{Hỏi-đáp theo khía cạnh trên đánh giá smartphone tiếng Việt} (\textit{Aspect-based Question Answering over Vietnamese smartphone reviews}).

\begin{itemize}[leftmargin=1.6cm,itemsep=1pt]
  \item \textbf{Đầu vào}: câu hỏi tiếng Việt $q$ về một khía cạnh sản phẩm (vd.\ ``Pin điện thoại Samsung có dùng được lâu không?''), và một kho $\mathcal{D}$ gồm $N$ đánh giá thực tế.
  \item \textbf{Đầu ra}: câu trả lời ngắn $a$ bằng tiếng Việt, \textbf{neo (grounded)} vào tập review liên quan $\mathcal{R}\subseteq\mathcal{D}$ được truy xuất, kèm thống kê cảm xúc theo khía cạnh từ Knowledge Graph.
  \item \textbf{Bài toán con cốt lõi} (được đánh giá định lượng): cho câu hỏi nhắm khía cạnh $\alpha$ (vd.\ \texttt{BATTERY}), \textbf{xếp hạng} $\mathcal{D}$ sao cho các review thực sự bàn về $\alpha$ nằm top đầu --- đo bằng Precision@$k$ và MRR.
\end{itemize}

\noindent\textbf{Vì sao khó?} (1) \textit{Lexical gap}: câu hỏi và review dùng từ ngữ khác nhau cho cùng khía cạnh; (2) \textit{đa nhãn}: một review thường nhắc nhiều khía cạnh (trung bình $>$2 aspect/review trong UIT-ViSFD); (3) \textit{tiếng Việt}: tách từ, từ ghép, viết tắt, OOV; (4) \textit{giảm bịa đặt}: câu trả lời phải bám dữ liệu thật, không suy diễn từ tham số LLM.

\subsection{Mục tiêu dự án}

\begin{enumerate}[leftmargin=2cm]
  \item Xây dựng hệ thống \textbf{Vietnamese Graph RAG} cho bài toán hỏi đáp về sản phẩm TMĐT.
  \item \textbf{So sánh} các phương pháp biểu diễn văn bản: TF-IDF, Word2Vec, GloVe, PhoBERT; có thêm subword (BPE) và mô hình tuần tự \textbf{BiLSTM}.
  \item \textbf{Trích xuất thực thể} (NER) và \textbf{phân loại aspect} bằng BiLSTM từ đánh giá.
  \item Xây dựng \textbf{Knowledge Graph} thể hiện quan hệ sản phẩm -- thuộc tính -- cảm xúc.
  \item \textbf{Đánh giá} hiệu quả Graph RAG (ablation Bi-encoder / Attention / Graph).
  \item \textbf{Đóng gói LLMOps}: serving API, observability, vòng lặp train$\to$deploy mô hình.
\end{enumerate}

\subsection{Phạm vi}

\begin{itemize}[leftmargin=2cm]
  \item \textbf{Domain}: Thương mại điện tử --- đánh giá sản phẩm tiếng Việt (smartphone + đa sản phẩm)
  \item \textbf{Dataset}: \textbf{(1)} UIT-ViSFD~\cite{luc2021visfd} --- 11,122 review smartphone, 10 aspect (train 7,786 dùng thực nghiệm); \textbf{(2)} Shopee reviews --- 3,154 review đa sản phẩm (26 sản phẩm, 24 shop) có \texttt{product/shop/rating}. Tổng corpus truy xuất $\approx$ 10,9k review.
  \item \textbf{Mô hình}: TF-IDF, Word2Vec, GloVe-SVD (Lec02); Subword/BPE (Lec03); BiLSTM (Lec05); PhoBERT + Attention/MaxSim (Lec06)
  \item \textbf{LLM}: OpenAI API (gpt-4o-mini) cho phần Generation
  \item \textbf{Triển khai}: pipeline LLMOps (FastAPI + Gradio + observability + CI/Docker), có vòng lặp train$\to$deploy mô hình BiLSTM
\end{itemize}

\section{Cơ sở lý thuyết}
% ===================================================================

\subsection{Tiền xử lý văn bản tiếng Việt}

Tiếng Việt là ngôn ngữ \textbf{đơn lập} (isolating language), trong đó ranh giới từ không được xác định bằng khoảng trắng. Ví dụ: ``học sinh'' là một từ ghép, không phải hai từ riêng biệt.

\textbf{Công cụ sử dụng}: Thư viện \texttt{underthesea} cung cấp chức năng tách từ (word segmentation) cho tiếng Việt:

\begin{lstlisting}[caption={Tách từ tiếng Việt}]
from underthesea import word_tokenize
text = "Truong Dai hoc Khoa hoc Tu nhien"
tokens = word_tokenize(text, format="text")
# "Truong_Dai_hoc Khoa_hoc Tu_nhien"
\end{lstlisting}

\subsection{Bag of Words và TF-IDF}

\textbf{TF-IDF} (Term Frequency -- Inverse Document Frequency) đo lường mức độ quan trọng của một từ trong tài liệu so với toàn bộ tập tài liệu:
\[
  \text{TF-IDF}(t, d) = \text{TF}(t, d) \times \log\frac{N}{1 + \text{DF}(t)}
\]

Trong dự án, TF-IDF được sử dụng làm \textbf{baseline} cho việc truy xuất đánh giá sản phẩm liên quan đến câu hỏi của người dùng.

\subsection{Word Embeddings: Word2Vec và GloVe}

\textbf{Word2Vec}~\cite{mikolov2013} học biểu diễn từ dựa trên \textit{ngữ cảnh lân cận}:
\begin{itemize}
  \item \textbf{Skip-gram}: Dự đoán từ ngữ cảnh từ từ trung tâm
  \item \textbf{CBOW}: Dự đoán từ trung tâm từ ngữ cảnh
\end{itemize}

\textbf{GloVe}~\cite{pennington2014} học embeddings dựa trên \textit{ma trận đồng xuất hiện}, tối ưu hàm mất mát:
\[
  J = \sum_{i,j} f(X_{ij}) \left( w_i^T w_j + b_i + b_j - \log X_{ij} \right)^2
\]

\subsection{Mô hình Subword (BPE)}

Các embedding ở trên giả định một \textbf{từ vựng cố định}, nên khi gặp \textbf{từ hiếm hoặc ngoài từ vựng (OOV)} sẽ không có vector biểu diễn. \textbf{Byte-Pair Encoding (BPE)}~\cite{sennrich2016bpe} khắc phục bằng cách tách từ thành các \textbf{mảnh con (subword)} xuất hiện thường xuyên --- ví dụ ``chống\_nước'' $\to$ [``chống'', ``\_nước'']. Nhờ vậy mọi từ đều biểu diễn được qua tổ hợp subword, kể cả từ chưa từng thấy khi huấn luyện. PhoBERT sử dụng BPE trên dữ liệu tiếng Việt đã tách từ.

\subsection{Mạng nơ-ron hồi quy (RNN / LSTM)}

RNN xử lý dữ liệu \textbf{tuần tự} bằng trạng thái ẩn truyền qua từng bước thời gian: $h_t = f(h_{t-1}, x_t)$. \textbf{LSTM} bổ sung các cổng (input/forget/output) để giảm hiện tượng \textit{vanishing gradient} và nắm bắt phụ thuộc xa. \textbf{BiLSTM} đọc chuỗi theo cả hai chiều rồi ghép biểu diễn, phù hợp cho bài toán phân loại câu. Trong dự án, BiLSTM được dùng để phân loại đa nhãn aspect (Mục~\ref{sec:rnn}).

\subsection{Transformer và BERT/PhoBERT}

\textbf{Transformer}~\cite{vaswani2017} sử dụng cơ chế \textbf{Self-Attention} để xử lý song song toàn bộ chuỗi:
\[
  \text{Attention}(Q, K, V) = \text{softmax}\left(\frac{QK^T}{\sqrt{d_k}}\right) V
\]

\textbf{BERT}~\cite{devlin2019} mở rộng Transformer thành mô hình ngôn ngữ hai chiều pre-train quy mô lớn. \textbf{PhoBERT}~\cite{phobert2020} là phiên bản pre-train trên 20GB dữ liệu tiếng Việt, đạt state-of-the-art trên nhiều tác vụ NLP tiếng Việt; dự án dùng nó làm encoder chính.

\subsection{Retrieval-Augmented Generation (RAG)}

RAG~\cite{lewis2020rag} kết hợp hai thành phần:
\begin{enumerate}
  \item \textbf{Retriever}: Tìm kiếm tài liệu liên quan từ knowledge base
  \item \textbf{Generator}: Sinh câu trả lời dựa trên tài liệu đã tìm được
\end{enumerate}

\subsection{Knowledge Graph và Graph RAG}

Knowledge Graph lưu trữ thông tin dưới dạng \textbf{bộ ba} (triple): $(h, r, t)$ --- head entity, relation, tail entity. Ví dụ trong TMĐT:

\begin{center}
\texttt{(iPhone\_15, thuộc\_thương\_hiệu, Apple)} \\
\texttt{(iPhone\_15, có\_tính\_năng, Camera\_48MP)} \\
\texttt{(Camera\_48MP, được\_đánh\_giá, Tích\_cực\_85\%)}
\end{center}

Graph RAG~\cite{edge2024graphrag} bổ sung bước \textbf{graph traversal} vào pipeline RAG, cho phép truy xuất thông tin có cấu trúc --- ví dụ khi hỏi về ``camera iPhone 15'', hệ thống có thể duyệt graph để tìm tất cả đánh giá liên quan đến tính năng camera.

\subsection{Dataset chính: UIT-ViSFD}

\textbf{UIT-ViSFD} (Vietnamese Smartphone Feedback Dataset) là bộ dữ liệu đánh giá smartphone từ sàn TMĐT Việt Nam, được gán nhãn bởi con người với:

\begin{itemize}
  \item \textbf{10 aspect categories}: SCREEN, CAMERA, BATTERY, PERFORMANCE, STORAGE, DESIGN, PRICE, GENERAL, FEATURES, SER\&ACC (Service \& Accessories)
  \item \textbf{3 sentiment polarities}: Positive, Negative, Neutral
  \item \textbf{11,122 comments} với annotation ở mức aspect-level
\end{itemize}

\textbf{Ví dụ dữ liệu}:
\begin{quote}
\textit{``Camera chụp đêm rất nét, pin dùng được 2 ngày, nhưng giá hơi cao so với cấu hình.''}\\
$\rightarrow$ CAMERA: Positive, BATTERY: Positive, PRICE: Negative
\end{quote}

\begin{lstlisting}[caption={Tải và parse nhãn UIT-ViSFD (đúng định dạng thực tế)}]
import pandas as pd, re
df = pd.read_csv("data/raw/Train.csv")   # cot: index, comment, n_star, label
# nhan co dang: "{CAMERA#Positive};{SER&ACC#Negative};{OTHERS};"
PAT = re.compile(r"\{([\w&]+)#(\w+)\}")   # giu '&' de KHONG mat aspect SER&ACC
def parse_label(s):
    if pd.isna(s): return []
    return [(m.group(1), m.group(2)) for m in PAT.finditer(str(s))]
df["aspects"] = df["label"].apply(lambda s: [a for a, _ in parse_label(s)])
print(f"Total: {len(df)} | vi du nhan: {df['label'].iloc[0]}")
\end{lstlisting}

\subsection{Dataset bổ sung: Shopee reviews}

Để mở rộng ngoài domain smartphone, chúng tôi bổ sung \textbf{3.154 review Shopee đa sản phẩm} (26 sản phẩm, 24 shop) gồm các trường \texttt{comment, product\_name, shop\_name, rating\_star}; sau tiền xử lý, \textbf{3.137} bản được đưa vào kho truy xuất, nên tổng index $=7.786+3.137=$ \textbf{10.923} review. Bộ này \textbf{không có nhãn aspect}, nên được dùng cho hai mục đích: (1) mở rộng corpus truy xuất (Graph RAG trả lời được câu hỏi đa sản phẩm), và (2) cung cấp các thực thể \texttt{shop}/\texttt{product} cho Knowledge Graph. Aspect cho review Shopee được \textbf{dự đoán bằng mô hình BiLSTM} (Mục~\ref{sec:rnn}) thay vì nhãn vàng.

\subsection{Phân tích nhãn aspect}

Nhãn của UIT-ViSFD có dạng chuỗi \texttt{\{ASPECT\#Sentiment\}} ngăn cách bởi dấu \texttt{;}. Một điểm cần lưu ý: aspect \textbf{SER\&ACC} chứa ký tự \texttt{\&}, nên biểu thức chính quy ngây thơ \texttt{\textbackslash\{(\textbackslash w+)\#(\textbackslash w+)\textbackslash\}} sẽ \textbf{bỏ sót} aspect này (1.995 lần trong tập train --- aspect phổ biến thứ 7). Chúng tôi sửa thành \texttt{\textbackslash\{([\textbackslash w\&]+)\#(\textbackslash w+)\textbackslash\}} để giữ đủ 10 aspect.

\subsection{Pipeline tiền xử lý}

\begin{enumerate}
  \item \textbf{Cleaning}: lowercase, loại URL, ký tự đặc biệt, chữ số, chuẩn hóa khoảng trắng.
  \item \textbf{Word Segmentation}: tách từ tiếng Việt bằng \texttt{underthesea}.
  \item \textbf{Stopword Removal}: loại từ dừng (cho TF-IDF/Word2Vec/GloVe).
  \item \textbf{Lưu ý nhất quán}: PhoBERT dùng văn bản \textbf{đã tách từ nhưng giữ dấu/số} (không loại stopword), đúng với pipeline khuyến nghị của PhoBERT; các phương pháp đếm dùng bản đã làm sạch.
\end{enumerate}

\subsection{Module 1: Tiền xử lý văn bản tiếng Việt}
\label{sec:module1}

Tiếng Việt là ngôn ngữ đơn lập nên ranh giới từ không trùng khoảng trắng; chất lượng mọi biểu diễn downstream phụ thuộc bước tách từ. Hệ thống dùng \textbf{hai pipeline tiền xử lý song song}, chọn theo bản chất mô hình tiêu thụ đầu ra.

\noindent\textbf{(a) Pipeline ``đếm'' (TF-IDF / Word2Vec / GloVe-SVD / BiLSTM)} --- hàm \texttt{preprocess\_vietnamese} áp dụng tuần tự:
\begin{enumerate}[leftmargin=1.6cm,itemsep=0pt]
  \item Hạ chữ thường + cắt khoảng trắng thừa.
  \item Loại URL (\texttt{http\textbackslash S+|www\textbackslash S+}).
  \item Loại ký tự không phải chữ/số (\texttt{[\textasciicircum\textbackslash w\textbackslash s]}) --- bỏ dấu câu, emoji.
  \item \textbf{Chỉ loại số đứng riêng} (\texttt{\textbackslash b\textbackslash d+\textbackslash b}); \textit{giữ} token chữ--số mang nghĩa cấu hình (\texttt{64gb}, \texttt{120hz}, \texttt{5g}). Đây là quyết định thiết kế: xoá sạch chữ số sẽ làm mất tín hiệu domain điện thoại.
  \item Tách từ bằng \texttt{underthesea}.
  \item Loại \textbf{stopword} (tập 47 hư từ: \textit{và, của, là, có, cho, được, trong}\ldots) và token độ dài $\le 1$.
\end{enumerate}

\noindent\textbf{(b) Pipeline ``ngữ cảnh'' (PhoBERT)} --- \textit{chỉ} tách từ (\texttt{word\_tokenize}), \textbf{giữ nguyên dấu, số và stopword}, đúng khuyến nghị của PhoBERT (pre-train trên văn bản tách từ có dấu). Loại stopword/số ở đây sẽ phá vỡ ngữ cảnh mà self-attention khai thác.

\subsection{Module 2: So sánh Embedding Methods}
\label{sec:module2idx}

Đây là module quan trọng nhất, thể hiện kiến thức NLP cốt lõi. Chúng tôi so sánh 4 phương pháp biểu diễn văn bản:

  \centering
  \includegraphics[width=0.85\textwidth]{figures/embedding_comparison.png}
  \caption{So sánh Retrieval (MRR $\pm$ 95\% CI bootstrap): baseline Keyword + 4 phương pháp học, trên 100 truy vấn giàu từ khoá (trái) vs 16 truy vấn lexical-gap (phải). CI đã siết còn $\pm$0.05 nhưng vẫn chồng lấn ở truy vấn dễ; chỉ ở lexical-gap PhoBERT mới tách rõ.}
  \label{fig:embedding_comparison}

\textbf{Biểu diễn PhoBERT (dùng cho truy xuất).} Câu được tách từ rồi đưa qua PhoBERT; vector tài liệu là \textbf{trung bình có mặt nạ} của các trạng thái ẩn lớp cuối, sau đó $L_2$-chuẩn hoá để cosine quy về tích vô hướng:
\[
  \mathbf{e}(d)=\frac{\sum_{t} m_t\,\mathbf{h}_t}{\sum_t m_t},
  \qquad
  \hat{\mathbf{e}}(d)=\frac{\mathbf{e}(d)}{\lVert\mathbf{e}(d)\rVert_2},
\]
với $\mathbf{h}_t$ là embedding token $t$, $m_t\in\{0,1\}$ là mặt nạ chú ý ($\texttt{max\_seq\_len}=128$). Token-embedding cho MaxSim (Mục~\ref{sec:module5}) lấy tương tự nhưng giữ \textit{từng} token ($\text{max\_len}=64$). Cấu hình huấn luyện TF-IDF / Word2Vec (SGNS) / GloVe-SVD (PPMI) được trình bày định lượng ở Thí nghiệm~\ref{sec:exp1}.

\textbf{Lưu ý về độ đo}: không dùng ``trung bình cosine similarity'' để so sánh, vì độ lớn cosine \textbf{không so sánh được} giữa các không gian embedding khác nhau (TF-IDF thưa cho giá trị thấp hơn PhoBERT dày). Thay vào đó dùng \textbf{Precision@k} và \textbf{MRR} với ground-truth là nhãn aspect (xem Mục~\ref{sec:exp1}).

\textbf{Lưu ý về ``GloVe''}: do hạn chế tài nguyên, chúng tôi dùng \textbf{xấp xỉ kiểu GloVe} --- SVD trên ma trận đồng xuất hiện ở mức tài liệu (gần với LSA), không phải GloVe cửa sổ ngữ cảnh gốc.

\noindent\textbf{Vì sao $\tau_{\text{conf}}=30$?} Với ước lượng tỉ lệ $\hat p$ (vd.\ \% tích cực), nửa độ rộng khoảng tin cậy 95\% xấp xỉ $1.96\sqrt{\hat p(1-\hat p)/n}$. Tại $n=30$ (xấu nhất $\hat p{=}0.5$) sai số $\approx\pm0.18$; tại $n=126$ (pin Samsung) còn $\pm0.087$. Vậy $30$ là \textbf{sàn hợp lý}: dưới mức này \% cảm xúc dao động quá lớn nên hệ thống gắn cờ ``mẫu nhỏ'' thay vì khẳng định (hiện thực \texttt{confidence\_note}, dùng ở \texttt{pipeline.\_graph\_context}). Mức ``tối ưu'' của KG vì thế \textbf{bị chặn bởi dữ liệu}, không phải tham số: chỉ \textbf{47/120 ô} hãng$\times$aspect đạt $\tau_{\text{conf}}$ (xem Mục~\ref{sec:kg-data}).

\subsubsection*{(d) Tích hợp vào RAG (lúc truy vấn)}
Hàm \texttt{\_graph\_context} (\texttt{rag/pipeline.py}) biến KG thành ngữ cảnh cho LLM:
\begin{enumerate}[leftmargin=1.5cm,itemsep=0pt]
  \item $\alpha\gets$\texttt{aspect\_from\_query}($q$); \; $b\gets$\texttt{detect\_brand}($q$).
  \item Nếu $b$ xác định: $sd\gets$\texttt{graph\_query\_brand}($b,\alpha$); nếu rỗng $\to$ fallback \texttt{graph\_query}($\alpha$) toàn corpus.
  \item Tính \% cảm xúc từ trọng số cạnh; nếu $n<\tau_{\text{conf}}$ hoặc $\alpha{=}$STORAGE $\Rightarrow$ thêm cảnh báo độ tin cậy.
  \item \texttt{product\_context}($q$) cho câu hỏi đa sản phẩm (yêu cầu $\geq$2 token trùng).
  \item Ghép thống kê vào \textbf{prompt grounded} $\to$ LLM (Module 6).
\end{enumerate}

\subsection{Module 5: Truy xuất lai hai tầng với tái xếp hạng late-interaction}
\label{sec:module5}

Đây là thành phần lõi của ``Graph RAG''. Truy xuất gồm \textbf{hai tầng}: sinh ứng viên (dense $\cup$ sparse) để tối đa \textit{recall}, rồi tái xếp hạng bằng tổ hợp bốn tín hiệu để tối đa \textit{precision}.

\subsubsection*{(a) Tầng 1 --- Sinh ứng viên}
Cho truy vấn $q$ và corpus $\mathcal{D}$:
\begin{itemize}[leftmargin=1.5cm,itemsep=1pt]
  \item \textit{Dense} (PhoBERT bi-encoder): $\mathcal{C}_{\text{dense}}=\text{top-}N\big\{\cos(\hat{\mathbf{e}}(q),\hat{\mathbf{e}}(d))\big\}$ với $\hat{\mathbf{e}}$ là vector mean-pool đã $L_2$-chuẩn hoá (Mục~\ref{sec:module2idx}).
  \item \textit{Sparse} (BM25-Okapi~\cite{robertson2009bm25}): $\mathcal{C}_{\text{bm25}}=\text{top-}N\big\{\text{BM25}(q,d)\big\}$ --- bắt \textbf{khớp chính xác} (mã model, ``120hz'') mà dense dễ bỏ sót.
  \item \textbf{Hợp}: $\mathcal{C}=\mathcal{C}_{\text{dense}}\cup\mathcal{C}_{\text{bm25}}$ (tối đa $2N$ ứng viên), $N=50$.
\end{itemize}

\subsubsection*{(b) Tầng 2 --- Tái xếp hạng bốn tín hiệu}
Với mỗi $d\in\mathcal{C}$ tính bốn điểm thành phần:
\begin{align*}
  s_{\text{bi}}(d)    &= \cos(\hat{\mathbf{e}}(q),\hat{\mathbf{e}}(d)),
  & s_{\text{bm25}}(d) &= \text{BM25}(q,d), \\
  s_{\text{attn}}(d)  &= \text{MaxSim}(q,d)=\frac{1}{|q|}\sum_{i\in q}\max_{j\in d}\,\mathbf{t}^{q}_i\!\cdot\!\mathbf{t}^{d}_j,
  & s_{\text{graph}}(d)&= \mathbf{1}\!\left[\alpha(q)\in\mathcal{A}(d)\right].
\end{align*}
Trong đó $\mathbf{t}^{q},\mathbf{t}^{d}$ là embedding token đã $L_2$-chuẩn hoá (tích vô hướng $=$ cosine); MaxSim là \textbf{late-interaction kiểu ColBERT}~\cite{khattab2020colbert}, \textit{không tham số và không cần huấn luyện}. $\alpha(q)$ là aspect suy ra từ truy vấn (khớp từ khoá), $\mathcal{A}(d)$ là tập aspect \textit{quan sát trong nội dung} $d$ --- \textbf{độc lập với nhãn vàng dùng để chấm}, nên graph boost \textbf{không rò rỉ metric} (đây là điểm mấu chốt cho tính hợp lệ của TN3).

\subsubsection*{(c) Chuẩn hoá \& kết hợp}
Mỗi thành phần liên tục được min--max chuẩn hoá về $[0,1]$ trên tập $\mathcal{C}$, $\tilde{s}=(s-\min)/(\max-\min)$, để các thang đo khác nhau \textit{so sánh được} trước khi cộng tuyến tính:
\[
  \text{score}(d)=w_{\text{bi}}\,\tilde{s}_{\text{bi}}(d)
  + w_{\text{attn}}\,\tilde{s}_{\text{attn}}(d)
  + w_{\text{graph}}\,s_{\text{graph}}(d)
  + w_{\text{bm25}}\,\tilde{s}_{\text{bm25}}(d),
\]
trả về top-$k$ theo $\text{score}$.

\begin{algorithm}[H]
\caption{HybridRetrieve --- truy xuất lai 2 tầng (\texttt{rag/retrieval.py})}
\begin{algorithmic}[1]
\Require truy vấn $q$, corpus $\mathcal{D}$, trọng số $(w_{\text{bi}},w_{\text{attn}},w_{\text{graph}},w_{\text{bm25}})$
\State $\mathcal{C}_{\text{dense}}\gets\text{top-}N$ theo $\cos(\hat{\mathbf{e}}(q),\hat{\mathbf{e}}(d))$;\quad $\mathcal{C}_{\text{bm25}}\gets\text{top-}N$ theo $\text{BM25}(q,d)$
\State $\mathcal{C}\gets\mathcal{C}_{\text{dense}}\cup\mathcal{C}_{\text{bm25}}$
\For{$d\in\mathcal{C}$}
  \State $s_{\text{bi}},s_{\text{bm25}}\gets$ cosine, BM25 \Comment{tính cho toàn $\mathcal{C}$}
  \State $s_{\text{attn}}\gets \textsc{MaxSim}(q,d)$ nếu $d\in\mathcal{C}_{\text{dense}}$, ngược lại $0$ \Comment{token-encode tốn kém}
  \State $s_{\text{graph}}\gets\mathbf{1}[\alpha(q)\in\mathcal{A}(d)]$
\EndFor
\State min--max chuẩn hoá $s_{\text{bi}},s_{\text{attn}},s_{\text{bm25}}$ trên $\mathcal{C}$
\State \Return top-$k$ theo $w_{\text{bi}}\tilde{s}_{\text{bi}}+w_{\text{attn}}\tilde{s}_{\text{attn}}+w_{\text{graph}}s_{\text{graph}}+w_{\text{bm25}}\tilde{s}_{\text{bm25}}$
\end{algorithmic}
\end{algorithm}

\centering\small
\caption{Tham số truy xuất (\texttt{config.yaml}).}
\begin{tabular}{lll}
\toprule
\textbf{Tham số} & \textbf{Giá trị} & \textbf{Vai trò} \\
\midrule
$N$ (\texttt{n\_candidates}) & 50 & số ứng viên mỗi nhánh dense/sparse \\
$k$ (\texttt{top\_k}) & 5 & số review đưa sang Generation \\

\noindent\textbf{Tối ưu chi phí}: MaxSim phải mã hoá token nên (i) \textit{chỉ} tính cho $\mathcal{C}_{\text{dense}}$ (gán $0$ cho ứng viên chỉ-BM25), và (ii) memoize embedding token theo \texttt{doc\_idx} --- các ứng viên phổ biến lặp lại nhiều truy vấn chỉ encode một lần. Trọng số trên là giá trị tune-trên-dev (Mục~\ref{sec:metric-crit}), không tune trên tập báo cáo.

\subsection{Module 6: Sinh câu trả lời có kiểm soát (grounded generation)}
\label{sec:module6}

Mục tiêu: sinh câu trả lời tiếng Việt \textbf{neo chặt vào bằng chứng} (top-$k$ review + thống kê Knowledge Graph) nhằm \textit{giảm bịa đặt}.

\textbf{Dựng ngữ cảnh.} Cho tập review $\mathcal{R}$ (top-$k$, Mục~\ref{sec:module5}) và chuỗi thống kê KG $g$ (Mục~\ref{sec:module4}), prompt được ghép từ ba khối: (i) mỗi review cắt còn 240 ký tự, gắn tiền tố nguồn \texttt{[UIT-ViSFD|Shopee]}; (ii) khối thống kê KG; (iii) câu hỏi.

\textbf{Chính sách kiểm soát hai lớp.} (1) \textit{Grounding} --- chỉ thị ``CHỈ dựa trên dữ liệu dưới đây''. (2) \textit{Abstention} --- nếu dữ liệu không chứa thông số kỹ thuật được hỏi (IP68, Hz, GHz, mAh, gram, số năm bảo hành, phút sạc, độ phân giải) thì \textbf{bắt buộc} trả ``Đánh giá không đề cập thông tin này'', tuyệt đối không suy ra từ tham số mô hình.

\subsection{Độ đo (định nghĩa hình thức)}
\label{sec:metrics}

Đặt $Q$ là tập truy vấn, $\mathrm{rel}_q(d)\in\{0,1\}$ là phán quyết liên quan, $\mathrm{rank}_q$ là vị trí review liên quan \textit{đầu tiên}.

\noindent\textbf{Truy xuất}:
\[
  \text{P@}k=\frac{1}{|Q|}\sum_{q\in Q}\frac{1}{k}\sum_{i=1}^{k}\mathrm{rel}_q(d_i),
  \qquad
  \text{MRR}=\frac{1}{|Q|}\sum_{q\in Q}\frac{1}{\mathrm{rank}_q},
\]
\[
  \text{DCG@}k=\sum_{i=1}^{k}\frac{\mathrm{rel}_q(d_i)}{\log_2(i+1)},
  \qquad
  \text{NDCG@}k=\frac{\text{DCG@}k}{\text{IDCG@}k}\in[0,1].
\]

\noindent\textbf{Phân loại aspect (đa nhãn)}, với tập aspect $A$ ($|A|=10$):
\[
  \text{F1}_{\text{micro}}=\frac{2\sum_{a}TP_a}{2\sum_a TP_a+\sum_a FP_a+\sum_a FN_a},
  \qquad
  \text{F1}_{\text{macro}}=\frac{1}{|A|}\sum_{a\in A}\text{F1}_a.
\]
Micro gộp mọi nhãn (bị aspect phổ biến thống trị); macro trung bình đều theo aspect (phơi bày lớp hiếm) --- lý do báo cáo \textit{cả hai} (Mục~\ref{sec:rnn}).

\noindent\textbf{Chất lượng câu trả lời (TN5)}: \textit{faithfulness} $\in[0,100]$ (LLM-judge: \% nội dung câu trả lời được bằng chứng review$+$KG chứng minh); \textit{relevance} $\in\{1,\dots,5\}$;
\[
  \text{numeric-grounding}=\frac{\#\{\text{số trong câu trả lời có trong bằng chứng}\}}{\#\{\text{số trong câu trả lời}\}},
  \quad
  \text{abstention}=\frac{\#\{\text{câu ngoài phạm vi bị từ chối}\}}{\#\{\text{câu ngoài phạm vi}\}}.
\]

\subsection{Quy trình tái lập}
\label{sec:repro}

\subsection{Thí nghiệm 1: So sánh Embedding Methods}
\label{sec:exp1}

\textbf{Thiết lập}: \textbf{100 truy vấn phủ đủ 10 aspect} (10 câu/aspect; mở rộng từ 8$\to$25$\to$100 để \textbf{siết khoảng tin cậy}, dùng \textbf{cùng bộ truy vấn với Thí nghiệm~3}). Pool xếp hạng là \textbf{toàn corpus 10.923 review} (7.786 UIT-ViSFD có nhãn vàng + 3.137 review Shopee đóng vai nhiễu, không liên quan). Một review được coi là \textit{liên quan} nếu nhãn vàng của nó chứa aspect đó. Báo cáo Precision@5, Precision@10, MRR, kèm \textbf{khoảng tin cậy 95\% bootstrap} (2.000 lần resample truy vấn) để biết chênh lệch có vượt nhiễu không. Cùng một phán quyết liên quan áp dụng cho mọi phương pháp $\Rightarrow$ công bằng. Ta bổ sung \textbf{baseline ``Keyword''} --- xếp hạng theo \textit{số từ trùng lặp thô} giữa truy vấn và review (mô phỏng ô tìm kiếm từ khoá của sàn TMĐT, \textbf{không} ngữ nghĩa, \textbf{không} IDF) --- để đối chiếu trực tiếp ``không dùng học biểu diễn'' với ``dùng''. Word2Vec huấn luyện skip-gram + negative sampling (PyTorch); GloVe-SVD lấy từ SVD ma trận PPMI đồng xuất hiện; PhoBERT dùng vector mean-pool.

Lưu ý thêm: mọi CI vẫn còn rộng (16 truy vấn) nên các kết luận cần được xác nhận lại trên bộ lớn hơn.

\subsection{Thí nghiệm 5: ``KHÔNG dùng'' vs ``DÙNG'' ở mức câu trả lời}
\label{sec:exp5}

Các thí nghiệm trên đo \textit{retrieval}. Nhưng sản phẩm cuối là \textbf{câu trả lời}, và P@k/MRR \textbf{không} đo được câu trả lời có đúng/có bịa hay không. Thí nghiệm này trả lời thẳng câu hỏi cốt lõi của một bài RAG: \textit{dùng hệ thống có hơn không dùng không, và hơn bao nhiêu?} Ta trả lời \textbf{cùng bộ câu hỏi bằng 3 chế độ}:

\begin{itemize}[leftmargin=1.5cm,itemsep=1pt]
  \item \textbf{No-RAG} (closed-book): LLM trả lời \textit{tay không} --- không review, không Knowledge Graph. Đại diện ``\textbf{không dùng} hệ thống''.
  \item \textbf{Vanilla-RAG}: chỉ PhoBERT bi-encoder Top-5, \textbf{không} graph, \textbf{không} BM25/MaxSim.
  \item \textbf{Graph-RAG} (đầy đủ): hybrid retrieve + thống kê Knowledge Graph --- hệ thống của dự án.
\end{itemize}

\subsection{Đánh giá ý nghĩa thực tiễn}

Quy chiếu các con số về \textbf{bài toán thực} (Mục~\ref{sec:problem}):

\begin{itemize}[leftmargin=1.5cm,itemsep=1pt]
  \item \textbf{Hữu dụng so với ``đoán mò''}: MRR nhảy từ 0.416 (ngẫu nhiên) lên 0.845 (Graph RAG) --- nghĩa là review đúng khía cạnh gần như luôn nằm ở \textbf{1--2 vị trí đầu}, người dùng không phải cuộn qua hàng trăm bình luận.
  \item \textbf{Graph boost giải đúng nỗi đau ``tổng hợp xu hướng''}: ngoài việc xếp hạng review, KG trả về thống kê (vd.\ \texttt{BATTERY} 56\% tích cực / 34\% tiêu cực trên 3.604 lượt nhắc) --- thứ mà tìm kiếm từ khoá của sàn TMĐT không làm được.
  \item \textbf{Giảm bịa đặt (grounding) --- nay có SỐ}: faithfulness câu trả lời tăng từ 47.3 (No-RAG) lên \textbf{82.2} (Graph-RAG, Mục~\ref{sec:exp5}); harness numeric-grounding cho \textbf{100\% (9/9) con số} trong câu trả lời Graph-RAG truy được về bằng chứng (không bịa số). Đây là dùng \textit{hệ thống} thay vì để LLM bịa từ tham số.
  \item \textbf{Chi phí \& tốc độ chấp nhận được}: $\approx$ 4--7 giây/câu (gồm gọi LLM), chi phí token ghi log theo từng truy vấn --- khả thi cho một trợ lý hỏi đáp thời gian thực.
\end{itemize}

\noindent\textbf{So với hệ thống thực tế}: đây là một \textit{prototype quy mô lớp học} (corpus 10.9k review, 1 domain smartphone), chưa phải sản phẩm thương mại. Để triển khai thật cần: chống review giả, cập nhật real-time khi có review mới, và mở rộng đa domain --- xem Hướng phát triển.

\subsection{Bàn về tính đúng đắn của metrics \& dữ liệu đánh giá}
\label{sec:metric-crit}

Phần này trả lời trực tiếp hai câu hỏi phản biện: \textit{``metrics đánh giá có đúng không?''} và \textit{``nguồn dữ liệu có nhỏ quá không?''}.

\textbf{(1) Vì sao dùng nhiều chỉ số (P@k, MRR, NDCG@5) thay vì một?} MRR chỉ thưởng vị trí của \textit{hit đầu tiên}, nên trên pool 10.923 review với aspect phổ biến, nó dễ bị \textbf{phóng đại}: ở TN1b, baseline Keyword đạt MRR 0.736 dù P@5 chỉ 0.538 --- chỉ cần một review đúng lọt top do trùng vài từ phụ. P@k đo độ chính xác top-k, NDCG@5 thưởng cả thứ hạng. \textbf{Phải đọc cùng nhau}. Ta cũng \textbf{không} dùng ``trung bình cosine similarity'' (vô nghĩa khi so giữa các không gian embedding khác chiều/độ thưa).

\textbf{(2) Khoảng tin cậy \& ý nghĩa thống kê --- ta đã mở rộng eval lên 100 câu để kiểm.} Nâng bộ truy vấn từ 8$\to$25$\to$\textbf{100 câu} siết CI của MRR từ $\pm$0.15 còn $\pm$0.05. \textbf{Nhưng kết luận trung thực càng rõ}: trên truy vấn \textit{giàu từ khoá}, ngay cả với 100 câu các CI \textbf{vẫn chồng lấn} (mọi MRR 0.76--0.89) --- các phương pháp embedding \textbf{không} tách biệt có ý nghĩa khi truy vấn dễ; khác biệt thật chỉ xuất hiện ở \textit{lexical gap} (TN1b, dựa P@k). \textit{Lưu ý phân biệt}: \textbf{corpus} 10.9k review \textit{đủ lớn} để làm kho truy xuất; cái còn thiếu là \textbf{truy vấn lexical-gap có nhãn} (TN1b mới 16 câu) --- hướng mở rộng tiếp theo.

\textbf{(3) Tách dev/test để tune trung thực (chống rò rỉ).} Ban đầu trọng số hybrid được grid-search trên \textit{chính} bộ truy vấn dùng để báo cáo --- đó là ``tune trên tập test''. Ta tách \textbf{50 dev / 50 test} (phân tầng theo aspect): tune trên dev rồi chấm test cho MRR \textbf{0.860}; nếu (sai) tune thẳng trên test thì MRR ``đẹp'' hơn \textbf{0.873}. \textbf{Chênh $\approx$0.013 là mức thổi phồng do overfit}. Đáng chú ý: mức này \textbf{giảm từ $\approx$0.05 (bộ 25 câu) còn $\approx$0.013 (bộ 100 câu)} --- xác nhận thêm rằng bộ eval càng nhỏ thì con số ``tuned'' càng bị thổi phồng. Con số nên báo cáo là loại trung thực (tune-trên-dev).

\textbf{(4) Retrieval-relevance $\neq$ answer-quality.} P@k/MRR chỉ đo ``review đúng aspect lọt top'', \textbf{không} đo câu trả lời có đúng/có bịa. Vì mục tiêu thật của bài toán là \textit{giảm bịa đặt}, ta bổ sung thước đo \textbf{ở mức câu trả lời}: faithfulness (TN5, Mục~\ref{sec:exp5}) và \textbf{numeric-grounding}: ở TN5, \textbf{100\% (9/9)} con số trong câu trả lời Graph-RAG truy được về bằng chứng (không bịa số). Một self-test tự động riêng (\texttt{scripts/eval\_grounding.py}, 8 câu grounded + 6 câu đòi thông số cứng) đo khả năng \textit{từ chối}: hệ thống từ chối \textbf{66.7\% (4/6)} câu đòi thông số không có trong dữ liệu, nhưng \textbf{vẫn để lọt} 2 số bịa (``bảo hành 1 năm'', ``nặng 200g'') --- xác nhận abstention còn chưa kín (xem Hạn chế). Đây là điểm khiến metric \textit{khớp với mục tiêu}, không chỉ khớp với tác vụ xếp hạng.

\textbf{Tóm lại}: hệ metric đã được \textit{sửa cho đúng hơn} (đa chỉ số, CI, dev/test, faithfulness/grounding), và chính các chỉ số đó \textbf{tự phơi bày} giới hạn còn lại (bộ eval nhỏ, judge tự thiên vị) --- xem Hạn chế.

\subsection{Tóm tắt kết quả}
Dự án xây dựng pipeline Graph RAG hoàn chỉnh cho đánh giá sản phẩm tiếng Việt trên \textbf{hai dataset} (UIT-ViSFD smartphone + Shopee đa sản phẩm), phủ trọn 6 lecture của môn. Đóng góp định lượng cốt lõi là \textbf{so sánh ``không dùng vs dùng'' ở cả hai mức}: (i) \textit{retrieval} --- baseline Keyword/TF-IDF so với học biểu diễn, và ablation bi-encoder $\to$ +MaxSim $\to$ +graph $\to$ +BM25 (Graph RAG đạt MRR cao nhất \textbf{0.845}, graph boost +0.033 robust); (ii) \textit{câu trả lời} --- No-RAG vs Vanilla-RAG vs Graph-RAG, faithfulness tăng \textbf{47.3 $\to$ 82.2} (TN5), lần đầu định lượng ``RAG giảm bịa đặt''. Đánh giá được làm \textbf{chặt về phương pháp}: đa chỉ số (P@k/MRR/NDCG@5), khoảng tin cậy bootstrap, mở rộng eval lên 100 câu, tách dev/test (đo overfit do tuning $\approx$0.013 MRR), và đo grounding ở mức câu trả lời --- đồng thời \textbf{trung thực phơi bày} các điểm yếu (bộ eval nhỏ, STORAGE F1=0, judge tự thiên vị). Toàn bộ đóng gói thành \textbf{pipeline LLMOps} (serving API, 2 giao diện Gradio, observability chi phí, regression gate, Docker/CI) với \textbf{vòng lặp train$\to$deploy} BiLSTM khép kín.

\subsection{Hạn chế}
Các hạn chế được nhóm theo \textit{bản chất}; mỗi mục nêu \textbf{hạn chế $\to$ bước khắc phục $\to$ hiệu quả kỳ vọng}.

\smallskip\noindent\textbf{(a) Hạn chế về bằng chứng \& đánh giá}
\begin{itemize}[leftmargin=1.5cm,itemsep=2pt]
  \item \textbf{Bộ truy vấn đánh giá còn nhỏ}: với 16 truy vấn lexical-gap, CI của MRR rộng tới $\pm$0.17 nên \textit{chênh lệch dưới $\approx$0.15 MRR chưa kết luận được} (Mục~\ref{sec:metric-crit}). $\to$ \textit{Khắc phục}: $\geq$100 truy vấn lexical-gap, $\geq$2 người soạn, đo độ đồng thuận (Cohen's $\kappa$). $\to$ \textit{Kỳ vọng}: siết CI còn $\pm$0.05, đủ tách PhoBERT khỏi nhóm từ vựng một cách có ý nghĩa thống kê.
  \item \textbf{LLM-judge cùng họ generator} (gpt-4o-mini, TN5) $\Rightarrow$ rủi ro tự thiên vị; numeric-grounding chỉ là proxy, không bắt được paraphrase-hallucination. $\to$ \textit{Khắc phục}: chấm lại bằng judge \textbf{khác họ} + \textbf{30 câu người chấm tay}, báo cáo độ đồng thuận người$\leftrightarrow$judge. $\to$ \textit{Kỳ vọng}: tách thiên vị khỏi tín hiệu thật, xác nhận xu hướng 47.3$\to$82.2 là thực.
  \item \textbf{Abstention chưa kín và chưa ổn định}: Graph-RAG chỉ từ chối \textbf{12.5\%} câu ngoài phạm vi (TN5), còn self-test \texttt{eval\_grounding.py} đạt \textbf{66.7\% (4/6)} nhưng vẫn để lọt số bịa (``200g'', ``bảo hành 1 năm''); chênh lệch lớn cho thấy abstention phụ thuộc mạnh vào bộ câu hỏi ($n$ nhỏ). $\to$ \textit{Khắc phục}: chính sách abstention hiệu chỉnh (ngưỡng tin cậy + mẫu câu ``không có thông số'') và benchmark out-of-scope chuyên biệt. $\to$ \textit{Kỳ vọng}: từ chối ổn định, không bịa thông số cứng.
\end{itemize}

\smallskip\noindent\textbf{(b) Hạn chế về phương pháp}
\begin{itemize}[leftmargin=1.5cm,itemsep=2pt]
  \item \textbf{Phân loại aspect mất cân bằng}: STORAGE F1=0, SCREEN recall 0.40 (micro-F1 0.79 che giấu, lộ ra ở macro-F1 0.68). $\to$ \textit{Khắc phục}: class weighting / focal loss / oversampling / fine-tune PhoBERT. $\to$ \textit{Kỳ vọng}: macro-F1 $>$0.8, cứu được lớp hiếm STORAGE.
  \item \textbf{Đóng góp ``+graph'' (+0.033 MRR) là ước lượng lạc quan}: graph boost dựa trên aspect \textit{quan sát} còn ground-truth dùng aspect \textit{nhãn vàng} --- khác nguồn nhưng \textbf{cùng ước lượng một aspect tiềm ẩn} nên vẫn tương quan nội tại. $\to$ \textit{Khắc phục}: đo boost với aspect dự đoán từ mô hình huấn luyện trên dữ liệu \textit{tách rời}, hoặc chỉ trên truy vấn có predicted$\neq$gold. $\to$ \textit{Kỳ vọng}: tách phần đóng góp thực của graph khỏi tương quan aspect.
  \item \textbf{Nhiều thành phần còn xấp xỉ/thủ công}: ``GloVe'' chỉ là SVD đồng xuất hiện mức tài liệu; MaxSim không huấn luyện; ánh xạ câu hỏi$\to$aspect bằng từ khóa thủ công; NER tổng quát + gazetteer (chưa có NER huấn luyện riêng domain). $\to$ \textit{Khắc phục}: GloVe cửa sổ ngữ cảnh gốc, fine-tune cross-encoder PhoBERT, phân loại aspect câu hỏi bằng mô hình, NER domain. $\to$ \textit{Kỳ vọng}: bớt phụ thuộc heuristic, phủ brand/aspect rộng và chính xác hơn.
\end{itemize}

\smallskip\noindent\textbf{(c) Hạn chế về dữ liệu \& phạm vi}
\begin{itemize}[leftmargin=1.5cm,itemsep=2pt]
  \item \textbf{Đa domain mới ở mức index}: review Shopee không có nhãn aspect vàng nên chỉ đóng vai nhiễu trong eval, chưa được \textit{đánh giá} như đáp án; aspect Shopee do BiLSTM \textit{dự đoán} (macro-F1 0.68, riêng STORAGE F1=0) $\Rightarrow$ nhãn nhiễu đáng kể lan truyền vào KG. $\to$ \textit{Khắc phục}: gán nhãn aspect vàng cho mẫu Shopee + gắn cờ tin cậy rộng cho thống kê Shopee. $\to$ \textit{Kỳ vọng}: đo được chất lượng đa domain thật sự, KG Shopee đáng tin hơn.
  \item \textbf{Chỉ một domain có nhãn vàng} (smartphone UIT-ViSFD): kết luận định lượng chỉ chắc trong domain này. $\to$ \textit{Khắc phục}: thu nhãn vàng cho domain khác. $\to$ \textit{Kỳ vọng}: kiểm chứng khả năng tổng quát hoá.
  \item \textbf{Chưa chống review giả} --- đe doạ trực tiếp giá trị ``tổng hợp xu hướng'' (thống kê trên review giả sẽ sai). $\to$ \textit{Khắc phục}: thêm phát hiện review giả/spam trước khi dựng KG. $\to$ \textit{Kỳ vọng}: bảo vệ chính giá trị cốt lõi của hệ thống.
\end{itemize}

\subsection{Hướng phát triển}
\begin{itemize}
  \item \textbf{Mở rộng bộ đánh giá $\geq$100 truy vấn} + kiểm định ý nghĩa thống kê; thay LLM-judge tự thiên vị bằng người chấm / judge khác họ.
  \item \textbf{Sửa mất cân bằng aspect} (class weight / focal loss / fine-tune PhoBERT, kỳ vọng macro-F1 $>$0.8); cải thiện abstention cho câu hỏi đòi thông số cứng.
  \item Fine-tune cross-encoder PhoBERT cho re-ranking với nhãn liên quan tách biệt train/test; phân loại aspect câu hỏi bằng mô hình thay từ khóa.
  \item Mở rộng KG (quan hệ sản phẩm--linh kiện, entity linking), gán nhãn aspect vàng cho Shopee để \textit{đánh giá} đa domain thật sự, và thêm phát hiện review giả.
\end{itemize}
